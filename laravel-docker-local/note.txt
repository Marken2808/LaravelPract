# migrate
>_ docker-compose exec php php /var/www/html/artisan migrate

# create model
>_ docker-compose exec php php /var/www/html/artisan make:model ${model_name} --migration

# create migration
>_ docker-compose exec php php /var/www/html/artisan make:migration create_${migration_relationship}_table

# seed
>_ docker-compose exec php php /var/www/html/artisan db:seed

# create factory
>_ docker-compose exec php php /var/www/html/artisan make:factory ${name_factory} --model=${name}

---

# run
>_ docker-compose run --rm
>_ // + composer require ${package}
>_ // + npm install && npm run dev
>_ // + artisan migrate
